314 numeric_matrix_dynamic_t RRE() const {
315 numeric_matrix_dynamic_t m(*this);
316 size_t lead = 0;
317 for (size_t r = 0; r < r_; ++r) {
318 if (lead >= c_) return m;
319 size_t i;
320 for (i = r; m[i][lead] == 0;) {
321 ++i;
322 if (i == r_) {
323 i = r;
324 ++lead;
325 if (lead == c_) return m;
326 }
327 }
328 m[i].swap(m[r]);
329 T val = m[r][lead];
330 for (size_t j = 0; j < c_; ++j) m[r][j] /= val;
331 for (i = 0; i < r_; ++i) {
332 if (i != r) {
333 val = m[i][lead];
334 for (size_t j = 0; j < c_; ++j)
335 m[i][j] -= val * m[r][j];
336 }
337 }
338 ++lead;
339 }
340 m.fix_zero();
341 return m;
342 }

344 numeric_matrix_dynamic_lu_t<T> LU() const {
345 size_t n = vt::min(r_, c_);
346 numeric_matrix_dynamic_t lower(n, n);
347 numeric_matrix_dynamic_t upper(n, n);
348 for (size_t i = 0; i < n; ++i) {
349 for (size_t k = 0; k < n; ++k) {
350 T sum_ = 0;
351 for (size_t j = 0; j < i; ++j) sum_ += lower[i][j] * upper[j][k];
352 upper[i][k] = vector_[i][k] - sum_;
353 }
354 for (size_t k = i; k < n; ++k) {
355 if (i == k) lower[i][i] = 1;
356 else {
357 T sum_ = 0;
358 for (size_t j = 0; j < i; ++j) sum_ += lower[k][j] * upper[j][i];
359 lower[k][i] = (vector_[k][i] - sum_) / upper[i][i];
360 }
361 }
362 }
363 return {lower, upper};
364 }

469 static numeric_matrix_dynamic_t &mm_strassen(numeric_matrix_dynamic_t &C, const numeric_matrix_dynamic_t &A, const numeric_matrix_dynamic_t &B) {
470 size_t rA = closest_2(A.r_);
471 size_t cA = closest_2(A.c_);
472 size_t rB = closest_2(B.r_);
473 size_t cB = closest_2(B.c_);
474
475 numeric_matrix_dynamic_t A11, A12, A21, A22, B11, B12, B21, B22;
476
477 if (rA == A.r_ && cA == A.c_ && rB == B.r_ && cB == B.c_) {
478 A11 = vt::move(A.slice(0, 0, rA / 2, cA / 2));
479 A12 = vt::move(A.slice(0, cA / 2, rA / 2, cA));
480 A21 = vt::move(A.slice(rA / 2, 0, rA, cA / 2));
481 A22 = vt::move(A.slice(rA / 2, cA / 2, rA, cA));
482 B11 = vt::move(B.slice(0, 0, rB / 2, cB / 2));
483 B12 = vt::move(B.slice(0, cB / 2, rB / 2, cB));
484 B21 = vt::move(B.slice(rB / 2, 0, rB, cB / 2));
485 B22 = vt::move(B.slice(rB / 2, cB / 2, rB, cB));
486 } else {
487 numeric_matrix_dynamic_t Ap(rA, cA);
488 Ap.insert(A);
489 numeric_matrix_dynamic_t Bp(rB, cB);
490 Bp.insert(B);
491 A11 = vt::move(Ap.slice(0, 0, rA / 2, cA / 2));
492 A12 = vt::move(Ap.slice(0, cA / 2, rA / 2, cA));
493 A21 = vt::move(Ap.slice(rA / 2, 0, rA, cA / 2));
494 A22 = vt::move(Ap.slice(rA / 2, cA / 2, rA, cA));
495 B11 = vt::move(Bp.slice(0, 0, rB / 2, cB / 2));
496 B12 = vt::move(Bp.slice(0, cB / 2, rB / 2, cB));
497 B21 = vt::move(Bp.slice(rB / 2, 0, rB, cB / 2));
498 B22 = vt::move(Bp.slice(rB / 2, cB / 2, rB, cB));
499 C = vt::move(numeric_matrix_dynamic_t(rA, cB));
500 }
501
502 numeric_matrix_dynamic_t M1 = vt::move((A11 + A22) * (B11 + B22));
503 numeric_matrix_dynamic_t M2 = vt::move((A21 + A22) * B11);
504 numeric_matrix_dynamic_t M3 = vt::move(A11 * (B12 - B22));
505 numeric_matrix_dynamic_t M4 = vt::move(A22 * (B21 - B11));
506 numeric_matrix_dynamic_t M5 = vt::move((A11 + A12) * B22);
507 numeric_matrix_dynamic_t M6 = vt::move((A21 - A11) * (B11 + B12));
508 numeric_matrix_dynamic_t M7 = vt::move((A12 - A22) * (B21 + B22));
509 C.insert(0, 0, M1 + M4 - M5 + M7);
510 C.insert(0, C.c_ / 2, M3 + M5);
511 C.insert(C.r_ / 2, 0, M2 + M4);
512 C.insert(C.r_ / 2, C.c_ / 2, M1 - M2 + M3 + M6);
513
514 if (rA != A.r_ || cB != B.c_)
515 C = C.slice(0, 0, A.r_, B.c_);
516 return C;
517 }
