347 numeric_matrix_static_t<T, Order, Order> operator^(size_t n) {
348 static_assert(static_is_a_square_matrix(), "Non-square matrix can\'t use power operator.");
349 if (n == 0) return identity();
350 if (n == 1) return numeric_matrix_static_t(*this);
351 numeric_matrix_static_t<T, Order, Order> base(*this);
352 numeric_matrix_static_t<T, Order, Order> product(vt::move(identity()));
353 while (n > 0) {
354 if (n % 2 == 1) product.operator*=(base);
355 if (n > 1) base *= base;
356 n >>= 1;
357 }
358 return product;
359 }

515 T det_from_lu(const numeric_matrix_static_lu_t<T, Order> &lu) const {
516 T det = 1;
517 size_t r_swaps = 0;
518 for (size_t i = 0; i < Order; ++i) {
519 det *= lu.u()[i][i];
520 if (lu.u()[i][i] == 0) return 0;
521 if (lu.l()[i][i] != 1) r_swaps++;
522 }
523 return (r_swaps % 2 == 0) ? det : -det;
524 }

570 numeric_matrix_static_lu_t<T, Order> LU() const {
571 static_assert(static_is_a_square_matrix(), "Can only find LU decomposition of a square matrix.");
572 numeric_matrix_static_t<T, Order, Order> lower;
573 numeric_matrix_static_t<T, Order, Order> upper;
574 for (size_t i = 0; i < Order; ++i) {
575 for (size_t k = 0; k < Order; ++k) {
576 T sum_ = 0;
577 for (size_t j = 0; j < i; ++j) sum_ += lower[i][j] * upper[j][k];
578 upper[i][k] = vector_[i][k] - sum_;
579 }
580 for (size_t k = i; k < Order; ++k) {
581 if (i == k) lower[i][i] = 1;
582 else {
583 T sum_ = 0;
584 for (size_t j = 0; j < i; ++j) sum_ += lower[k][j] * upper[j][i];
585 lower[k][i] = (vector_[k][i] - sum_) / upper[i][i];
586 }
587 }
588 }
589 return {lower, upper};
590 }

598 numeric_matrix_static_t RRE() const {
599 numeric_matrix_static_t m(*this);
600 size_t lead = 0;
601 for (size_t r = 0; r < Row; ++r) {
602 if (lead >= Col) return m;
603 size_t i;
604 for (i = r; m[i][lead] == 0;) {
605 ++i;
606 if (i == Row) {
607 i = r;
608 ++lead;
609 if (lead == Col) return m;
610 }
611 }
612 m[i].swap(m[r]);
613 T val = m[r][lead];
614 for (size_t j = 0; j < Col; ++j) m[r][j] /= val;
615 for (i = 0; i < Row; ++i) {
616 if (i != r) {
617 val = m[i][lead];
618 for (size_t j = 0; j < Col; ++j)
619 m[i][j] -= val * m[r][j];
620 }
621 }
622 ++lead;
623 }
624 m.fix_zero();
625 return m;
626 }

915 template<size_t OSize>
916 static numeric_matrix_static_t<T, OSize, OSize> inv_lt(numeric_matrix_static_t<T, OSize, OSize> &L) {
917 static_assert(static_is_a_square_matrix(), "Can only inverse a square matrix.");
918 numeric_matrix_static_t<T, OSize, OSize> X;
919 for (size_t k = 0; k < OSize; ++k) {
920 X[k][k] = 1 / L[k][k];
921 for (size_t i = k + 1; i < OSize; ++i) {
922 T acc = 0;
923 for (size_t j = k; j < i; ++j)
924 acc -= L[i][j] * X[j][k];
925 X[i][k] = acc / L[i][i];
926 }
927 }
928 L = vt::move(X);
929 return L;
930 }

935 numeric_matrix_static_t<T, OSize, OSize> tmp = vt::move(U.transpose());
936 return inv_lt(tmp).transpose();
